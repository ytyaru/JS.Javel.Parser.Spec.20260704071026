# API

単一責任の原則（SRP）を意識した美しいクラス設計ですね！ご提示いただいた設計思想、非常に共感します。
ご懸念されている「各Tokenizerが個別に位置を返すと、並び替えるために全トークンをメモリ（配列）に溜める必要があり、ストリーミング（小分け返却）のメリットが消える」という問題は、コンパイラやパーサの実装における典型的な課題です。
これを解決するための結論から言うと、各要素（Blank, Fence, Block）が個別にテキスト全体をスキャンするのではなく、最上位の Tokenizer が「文字インデックス（カーソル）」を1つだけ持ち、各Tokenizerを状態（コンテキスト）に応じて切り替えながら、テキストを上から順に消費していく『状態遷移（State）パターン』を採用します。
これなら、テキストを一度しか走査（シングルパス）せず、配列によるソートも不要なため、10万字でも最小限のメモリで上から順にトークンを yield できます。
以下に、この課題をクリアし、ご提示のAPI構造を維持したアルゴリズムとコード例を提案します。
------------------------------
## アルゴリズムの核心：カーソルの一方向進行とTokenizerの委譲

   1. 最上位の Tokenizer が、現在読んでいる文字の位置 pos を保持します。
   2. ループ内では、まず「特殊なブロック（空行やフェンス）」の条件を満たすか各子Tokenizerに先頭の数文字だけチェックさせます。
   3. 条件にマッチしたTokenizerにパース権を委譲し、そのトークンが終わる位置まで pos を進めてトークンを yield します。
   4. どの特殊ブロックにもマッチしなければ、通常の「ブロック（段落や見出し）」として処理し、pos を進めます。

これにより、常にテキストの先頭から末尾に向かって順番にトークンが生成されます。
------------------------------
## 実装コード例
ご提示いただいた設計図（クラス名やメソッド名）をベースに、責任分離を保ったままシームレスに動くように落とし込んだコードです。

// ==========================================// javel-parser.js (最上位パーサ)// ==========================================class JavelParser {
    constructor(options) {
        this.options = options;
        this.tokenizer = new Tokenizer(options);
        this.lexer = new Lexer(options);
    }
    
    *parse(manuscript) {
        // 最上位のASTノード（Blockレベル要素）を一つずつ順次返却
        for (let token of this.tokenizer.tokenize(manuscript)) {
            yield this.lexer.lex(token);
        }
    }
}
// ==========================================// Tokenizers (トークン生成・位置制御の責任)// ==========================================class Tokenizer {
    constructor(options) {
        this.options = options;
        this.blankTokenizer = new BlankTokenizer(options);
        this.fenceTokenizer = new FenceTokenizer(options);
        this.blockTokenizer = new BlockTokenizer(options);
    }

    *tokenize(manuscript) {
        let pos = 0;
        const length = manuscript.length;

        while (pos < length) {
            // 1. 空行のチェックと処理
            let blankToken = this.blankTokenizer.tryTokenize(manuscript, pos);
            if (blankToken) {
                pos = blankToken.end; // カーソルを進める
                if (blankToken.yieldable) yield blankToken; // 設定次第でbrノード用に出力
                continue;
            }

            // 2. フェンスのチェックと処理
            let fenceToken = this.fenceTokenizer.tryTokenize(manuscript, pos);
            if (fenceToken) {
                pos = fenceToken.end;
                yield fenceToken;
                continue;
            }

            // 3. 上記以外はすべて通常のブロック（見出し、主題区切り、段落）
            // ブロックの終わり（次の空行かフェンス、または文末）までを切り出す
            let blockToken = this.blockTokenizer.tokenizeBlock(manuscript, pos, this);
            pos = blockToken.end;
            yield blockToken;
        }
    }
}
class BlankTokenizer {
    constructor(options) { this.options = options; }
    
    // 同期的にその位置が空行（改行の連続）かチェックし、トークンを返す
    tryTokenize(manuscript, pos) {
        let current = pos;
        let newlineCount = 0;
        
        // 改行コード（CRLF含む）や半角スペースのみの行をスキップ
        while (current < manuscript.length) {
            if (manuscript[current] === '\n') {
                newlineCount++;
                current++;
            } else if (manuscript[current] === '\r' && manuscript[current+1] === '\n') {
                newlineCount++;
                current += 2;
            } else if (manuscript[current] === ' ' || manuscript[current] === '\t') {
                current++; // 空白文字は読み飛ばす
            } else {
                break; // 地の文が始まったら終了
            }
        }

        if (newlineCount === 0) return null;

        // 3連続以上の改行を2連続にするなどのオプション制御
        const shouldYieldBr = newlineCount >= 3 && this.options?.insertBr;

        return {
            type: 'blank',
            start: pos,
            end: current,
            yieldable: shouldYieldBr,
            count: newlineCount
        };
    }
}
class FenceTokenizer {
    constructor(options) { this.options = options; }

    tryTokenize(manuscript, pos) {
        // 行頭チェック（posが0、または直前が改行である必要がある）
        if (pos > 0 && manuscript[pos - 1] !== '\n') return null;

        let current = pos;
        // フェンス開始記号（~ や ` など）をカウント
        const marker = manuscript[current];
        if (marker !== '`' && marker !== '~') return null; // 記号の限定

        let markerCount = 0;
        while (current < manuscript.length && manuscript[current] === marker) {
            markerCount++;
            current++;
        }
        if (markerCount < 3) return null; // 3文字未満はフェンスじゃない

        // フェンスの引数（メタデータ）を取得（改行まで）
        let argStart = current;
        while (current < manuscript.length && manuscript[current] !== '\n' && manuscript[current] !== '\r') {
            current++;
        }
        const info = manuscript.slice(argStart, current).trim();
        
        // 改行をまたぐ
        if (manuscript[current] === '\r') current++;
        if (manuscript[current] === '\n') current++;

        // フェンスの終了行を探す（ネスト対応）
        let contentStart = current;
        let contentEnd = current;
        let openFences = 1;

        while (current < manuscript.length) {
            // 行頭の判定
            if (manuscript[current - 1] === '\n') {
                let checkPos = current;
                let closeCount = 0;
                while (checkPos < manuscript.length && manuscript[checkPos] === marker) {
                    closeCount++;
                    checkPos++;
                }

                // 終了行の条件チェック
                if (closeCount === markerCount) {
                    // 直後が改行または文末か確認
                    if (checkPos === manuscript.length || manuscript[checkPos] === '\n' || manuscript[checkPos] === '\r') {
                        openFences--;
                        if (openFences === 0) {
                            contentEnd = current - 1; // 直前の改行までがコンテンツ
                            current = checkPos; // 閉じ記号の先頭へ
                            if (manuscript[current] === '\r') current++;
                            if (manuscript[current] === '\n') current++;
                            
                            return {
                                type: 'fence',
                                start: pos,
                                end: current,
                                info: info,
                                content: manuscript.slice(contentStart, contentEnd)
                            };
                        }
                    }
                } else if (closeCount > markerCount) {
                    // 開始記号より多い場合は、中でより深いフェンスが始まったとみなす（ネスト対応）
                    openFences++;
                }
            }
            current++;
        }

        // 閉じられずに文末に達した場合
        return {
            type: 'fence',
            start: pos,
            end: manuscript.length,
            info: info,
            content: manuscript.slice(contentStart, manuscript.length)
        };
    }
}
class BlockTokenizer {
    constructor(options) { this.options = options; }

    tokenizeBlock(manuscript, pos, masterTokenizer) {
        let current = pos;
        
        // 次の「空行」または「フェンス」が始まるか、文末に達するまで文字を進める
        while (current < manuscript.length) {
            // 行頭の場合のみ、次のブロック要素の開始（空行やフェンス）を予知する
            if (current === pos || manuscript[current - 1] === '\n') {
                if (masterTokenizer.blankTokenizer.tryTokenize(manuscript, current) ||
                    masterTokenizer.fenceTokenizer.tryTokenize(manuscript, current)) {
                    break; // 次の要素が始まるので、現在の段落ブロックをここで打ち切る
                }
            }
            current++;
        }

        const rawText = manuscript.slice(pos, current).trim();
        let blockSubtype = 'paragraph';

        // 行頭の記号によるサブタイプ判定
        if (rawText.startsWith('#')) {
            blockSubtype = 'heading';
        } else if (rawText === '***' || rawText === '---') {
            blockSubtype = 'thematicBreak';
        }

        return {
            type: 'block',
            subtype: blockSubtype,
            start: pos,
            end: current,
            content: rawText
        };
    }
}
// ==========================================// Lexers (構文解釈・ASTノード変換の責任)// ==========================================class Lexer {
    constructor(options) {
        this.options = options;
        this.fenceLexer = new FenceLexer(options);
        this.blockLexer = new BlockLexer(options);
    }

    lex(token) {
        switch (token.type) {
            case 'blank':
                return { type: 'BreakElement', count: token.count };
            case 'fence':
                return this.fenceLexer.lex(token);
            case 'block':
                return this.blockLexer.lex(token);
            default:
                throw new Error(`Unknown token type: ${token.type}`);
        }
    }
}
class FenceLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        // エスケープやフェンス引数に応じた内部メタデータのパース
        return {
            type: 'FenceBlock',
            info: token.info,
            value: token.content // 必要に応じて内部のエスケープを解除
        };
    }
}
class BlockLexer {
    constructor(options) {
        this.options = options;
        // インライン要素のパースは、ブロックが確定した「このタイミング」で走らせる
        this.inlineTokenizer = new InlineTokenizer(options);
    }

    lex(token) {
        // ブロックごとのASTノードを生成
        const node = {
            type: token.subtype === 'heading' ? 'Heading' : 
                  token.subtype === 'thematicBreak' ? 'ThematicBreak' : 'Paragraph',
            children: []
        };

        if (node.type === 'Heading') {
            // 例: ### 見出し -> レベル3
            const match = token.content.match(/^(#{1,6})\s*(.*)/);
            node.level = match ? match[1].length : 1;
            node.children = this.inlineTokenizer.tokenizeAndLex(match ? match[2] : token.content);
        } else if (node.type === 'Paragraph') {
            // 段落の中身をインラインパース
            node.children = this.inlineTokenizer.tokenizeAndLex(token.content);
        }

        return node;
    }
}
// ==========================================// InlineTokenizer (文字レベル走査・エスケープ・ルビの責任)// ==========================================class InlineTokenizer {
    constructor(options) { this.options = options; }

    // インラインはブロックのテキスト（せいぜい数百文字）を対象にするため、
    // ここは配列で一気に返却してもメモリへの影響は皆無です。
    tokenizeAndLex(text) {
        const children = [];
        let i = 0;

        while (i < text.length) {
            // 1. エスケープ処理
            if (text[i] === '\\' && i + 1 < text.length) {
                children.push({ type: 'Text', value: text[i + 1] });
                i += 2;
                continue;
            }

            // 2. ルビ処理 (例: ｜漢字《かんじ》)
            if (text[i] === '｜' || text[i] === '|') {
                const endMark = text.indexOf('》', i);
                const startRuby = text.indexOf('《', i);
                if (startRuby !== -1 && endMark !== -1 && startRuby < endMark) {

const target = text.slice(i + 1, startRuby);
const ruby = text.slice(startRuby + 1, endMark);
children.push({ type: 'Ruby', target, ruby });
i = endMark + 1;
continue;
}
}
// 3. 強調 em (例: ＊強調＊)
if (text[i] === '＊' || text[i] === '*') {
const endEm = text.indexOf(text[i], i + 1);
if (endEm !== -1) {
children.push({ type: 'Emphasis', value: text.slice(i + 1, endEm) });
i = endEm + 1;
continue;
}
}
// 4. 通常のテキスト（1文字ずつ、またはまとめて追加）
let char = text[i];
if (children.length > 0 && children[children.length - 1].type === 'Text') {
children[children.length - 1].value += char;
} else {
children.push({ type: 'Text', value: char });
}
i++;
}
return children;
}
}

### このアルゴリズムが優れている理由

1. **メモリ消費量が最小（O(1)に近いバッファ）**
   テキストを走査するカーソル（`pos`）は常に**前から後ろへ一方通行**で進みます。配列にトークンを溜めてソートする必要が一切ないため、10万字の文字列があってもメモリを圧迫しません。
2. **完全な責任分離（単一責任の原則）**
   * `Tokenizer` 系列：文字列の「切り出し（どこからどこまでか）」の責任。
   * `Lexer` 系列：切り出された文字列を「意味のあるAST（構文木ノード）」に翻訳する責任。
   * `InlineTokenizer`：ブロック内部の細かい文字装飾（ルビやエスケープ）をパースする責任。
3. **拡張性**
   例えば新しく「割注」の構文を追加したい場合、ブロック全体の流れを壊すことなく、一番最後の `InlineTokenizer` の中に割注用の `if` 文（`〔 〕` などの判定）を追加するだけで対応できます。

この設計であれば、大規模な小説原稿でもブラウザやサーバーのメモリを犠牲にせず、サクサク高速にパースすることができます！

                    checkPos++;
                }

                // 終了行の条件チェック
                if (closeCount === markerCount) {
                    // 直後が改行または文末か確認
                    if (checkPos === manuscript.length || manuscript[checkPos] === '\n' || manuscript[checkPos] === '\r') {
                        openFences--;
                        if (openFences === 0) {
                            contentEnd = current - 1; // 直前の改行までがコンテンツ
                            current = checkPos; // 閉じ記号の先頭へ
                            if (manuscript[current] === '\r') current++;
                            if (manuscript[current] === '\n') current++;
                            
                            return {
                                type: 'fence',
                                start: pos,
                                end: current,
                                info: info,
                                content: manuscript.slice(contentStart, contentEnd)
                            };
                        }
                    }
                } else if (closeCount > markerCount) {
                    // 開始記号より多い場合は、中でより深いフェンスが始まったとみなす（ネスト対応）
                    openFences++;
                }
            }
            current++;
        }

        // 閉じられずに文末に達した場合
        return {
            type: 'fence',
            start: pos,
            end: manuscript.length,
            info: info,
            content: manuscript.slice(contentStart, manuscript.length)
        };
    }
}

class BlockTokenizer {
    constructor(options) { this.options = options; }

    tokenizeBlock(manuscript, pos, masterTokenizer) {
        let current = pos;
        
        // 次の「空行」または「フェンス」が始まるか、文末に達するまで文字を進める
        while (current < manuscript.length) {
            // 行頭の場合のみ、次のブロック要素の開始（空行やフェンス）を予知する
            if (current === pos || manuscript[current - 1] === '\n') {
                if (masterTokenizer.blankTokenizer.tryTokenize(manuscript, current) ||
                    masterTokenizer.fenceTokenizer.tryTokenize(manuscript, current)) {
                    break; // 次の要素が始まるので、現在の段落ブロックをここで打ち切る
                }
            }
            current++;
        }

        const rawText = manuscript.slice(pos, current).trim();
        let blockSubtype = 'paragraph';

        // 行頭の記号によるサブタイプ判定
        if (rawText.startsWith('#')) {
            blockSubtype = 'heading';
        } else if (rawText === '***' || rawText === '---') {
            blockSubtype = 'thematicBreak';
        }

        return {
            type: 'block',
            subtype: blockSubtype,
            start: pos,
            end: current,
            content: rawText
        };
    }
}

class BlockLexer {
    constructor(options) {
        this.options = options;
        // インライン要素のパースは、ブロックが確定した「このタイミング」で走らせる
        this.inlineTokenizer = new InlineTokenizer(options);
    }

    lex(token) {
        // ブロックごとのASTノードを生成
        const node = {
            type: token.subtype === 'heading' ? 'Heading' : 
                  token.subtype === 'thematicBreak' ? 'ThematicBreak' : 'Paragraph',
            children: []
        };

        if (node.type === 'Heading') {
            // 例: ### 見出し -> レベル3
            const match = token.content.match(/^(#{1,6})\s*(.*)/);
            node.level = match ? match[1].length : 1;
            node.children = this.inlineTokenizer.tokenizeAndLex(match ? match[2] : token.content);
        } else if (node.type === 'Paragraph') {
            // 段落の中身をインラインパース
            node.children = this.inlineTokenizer.tokenizeAndLex(token.content);
        }

        return node;
    }
}

// InlineTokenizer (文字レベル走査・エスケープ・ルビの責任)

class InlineTokenizer {
    constructor(options) { this.options = options; }

    // インラインはブロックのテキスト（せいぜい数百文字）を対象にするため、
    // ここは配列で一気に返却してもメモリへの影響は皆無です。
    tokenizeAndLex(text) {
        const children = [];
        let i = 0;

        while (i < text.length) {
            // 1. エスケープ処理
            if (text[i] === '\\' && i + 1 < text.length) {
                children.push({ type: 'Text', value: text[i + 1] });
                i += 2;
                continue;
            }

            // 2. ルビ処理 (例: ｜漢字《かんじ》)
            if (text[i] === '｜' || text[i] === '|') {
                const endMark = text.indexOf('》', i);
                const startRuby = text.indexOf('《', i);
                if (startRuby !== -1 && endMark !== -1 && startRuby < endMark) {

                    const target = text.slice(i + 1, startRuby);
                    const ruby = text.slice(startRuby + 1, endMark);
                    children.push({ type: 'Ruby', target, ruby });
                    i = endMark + 1;
                    continue;
                }
            }
            // 3. 強調 em (例: ＊強調＊)
            if (text[i] === '＊' || text[i] === '*') {
                const endEm = text.indexOf(text[i], i + 1);
                if (endEm !== -1) {
                    children.push({ type: 'Emphasis', value: text.slice(i + 1, endEm) });
                    i = endEm + 1;
                    continue;
                }
            }
            // 4. 通常のテキスト（1文字ずつ、またはまとめて追加）
            let char = text[i];
            if (children.length > 0 && children[children.length - 1].type === 'Text') {
                children[children.length - 1].value += char;
            } else {
                children.push({ type: 'Text', value: char });
            }
            i++;
        }
        return children;
    }
}

## 2. Tokenizers (範囲の切り出しのみに特化。中身は見ない)
最上位の Tokenizer は、テキストを「フェンス」「2連続以上の改行（空行）」「それ以外の塊（ブロック）」の3つにしか分類しません。

class Tokenizer {
    constructor(options) {
        this.options = options;
        this.blankTokenizer = new BlankTokenizer(options);
        this.fenceTokenizer = new FenceTokenizer(options);
        this.blockTokenizer = new BlockTokenizer(options);
    }

    *tokenize(manuscript) {
        let pos = 0;
        const length = manuscript.length;

        while (pos < length) {
            // 1. 空行（改行の連続）の切り出し
            let blankToken = this.blankTokenizer.tryTokenize(manuscript, pos);
            if (blankToken) {
                pos = blankToken.end;
                if (blankToken.yieldable) yield blankToken; // 3連続改行のbr変換用
                continue;
            }

            // 2. フェンスの切り出し（記号の数と、閉じるまでの範囲だけを特定）
            let fenceToken = this.fenceTokenizer.tryTokenize(manuscript, pos);
            if (fenceToken) {
                pos = fenceToken.end;
                yield fenceToken;
                continue;
            }

            // 3. 通常のブロック（2連続改行、またはフェンスの手前までをただ切り出す）
            let blockToken = this.blockTokenizer.tokenizeBlock(manuscript, pos, this);
            pos = blockToken.end;
            yield blockToken;
        }
    }
}
class BlankTokenizer {
    constructor(options) { this.options = options; }
    
    tryTokenize(manuscript, pos) {
        let current = pos;
        let newlineCount = 0;

        // 改行と、行内・行頭のスペース（空白行）をまとめて消費
        while (current < manuscript.length) {
            if (manuscript[current] === '\n') {
                newlineCount++;
                current++;
            } else if (manuscript[current] === ' ' || manuscript[current] === '\t') {
                current++;
            } else {
                break;
            }
        }

        if (newlineCount === 0) return null;

        // 【ここで制御！】フェンス外の3連続以上の改行を設定次第で制限する
        // 例: 3連続以上でbr要素を詰めるオプションがある場合のみ、Lexerに渡すトークンを生成
        const shouldYield = newlineCount >= 3 && this.options?.adjustNewlines;

        return {
            type: 'blank',
            start: pos,
            end: current,
            count: newlineCount,
            yieldable: shouldYield
        };
    }
}
class FenceTokenizer {
    constructor(options) { this.options = options; }

    tryTokenize(manuscript, pos) {
        // 行頭チェック
        if (pos > 0 && manuscript[pos - 1] !== '\n') return null;

        let current = pos;
        const marker = manuscript[current];
        if (marker !== '`' && marker !== '~') return null;

        let markerCount = 0;
        while (current < manuscript.length && manuscript[current] === marker) {
            markerCount++;
            current++;
        }
        if (markerCount < 3) return null;

        // 引数部分（改行まで）
        let argStart = current;
        while (current < manuscript.length && manuscript[current] !== '\n') {
            current++;
        }
        const info = manuscript.slice(argStart, current).trim();
        if (manuscript[current] === '\n') current++;

        let contentStart = current;
        let contentEnd = current;
        let openFences = 1;

        // 終了行の探索（ネスト対応）
        while (current < manuscript.length) {
            if (manuscript[current - 1] === '\n') {
                let checkPos = current;
                let closeCount = 0;
                while (checkPos < manuscript.length && manuscript[checkPos] === marker) {
                    closeCount++;
                    checkPos++;
                }

                if (closeCount === markerCount) {
                    if (checkPos === manuscript.length || manuscript[checkPos] === '\n') {
                        openFences--;
                        if (openFences === 0) {
                            contentEnd = current - 1;
                            current = checkPos;
                            if (manuscript[current] === '\n') current++;
                            
                            return {
                                type: 'fence',
                                start: pos,
                                end: current,
                                info: info,
                                rawContent: manuscript.slice(contentStart, contentEnd)
                            };
                        }
                    }
                } else if (closeCount > markerCount) {
                    openFences++; // より深いフェンスの開始
                }
            }
            current++;
        }

        return {
            type: 'fence', start: pos, end: manuscript.length, info: info,
            rawContent: manuscript.slice(contentStart, manuscript.length)
        };
    }
}
class BlockTokenizer {
    constructor(options) { this.options = options; }

    tokenizeBlock(manuscript, pos, masterTokenizer) {
        let current = pos;
        
        while (current < manuscript.length) {
            // 新しい行が始まるタイミングで、次の特殊要素（空行やフェンス）がないか予知する
            if (current === pos || manuscript[current - 1] === '\n') {
                if (masterTokenizer.blankTokenizer.tryTokenize(manuscript, current) ||
                    masterTokenizer.fenceTokenizer.tryTokenize(manuscript, current)) {
                    break; // 次の要素が始まる直前でこのブロックを閉じる
                }
            }
            current++;
        }

        return {
            type: 'block',
            start: pos,
            end: current,
            rawText: manuscript.slice(pos, current)
        };
    }
}

## 3. Lexers (意味の解釈、詳細クラスへの委譲、ASTノードの生成)
最上位の Lexer は、トークンの type（フェンス、ブロック、空行）だけを見て詳細なLexerクラスへ丸投げします。中身を見て「見出し」だの「段落」だの判定するのは各詳細クラスの単一責任です。

class Lexer {
    constructor(options) {
        this.options = options;
        this.fenceLexer = new FenceLexer(options);
        this.blockLexer = new BlockLexer(options);
        this.brLexer = new BrLexer(options);
    }

    lex(token) {
        switch (token.type) {
            case 'blank':
                return this.brLexer.lex(token); // 連続改行によるbr要素ノード生成
            case 'fence':
                return this.fenceLexer.lex(token);
            case 'block':
                return this.blockLexer.lex(token);
            default:
                throw new Error(`Unknown token type: ${token.type}`);
        }
    }
}

    constructor(options) {

        this.headingLexer = new HeadingLexer(options);
        this.thematicBreakLexer = new ThematicBreakLexer(options);

## 4. Inline 系統の独立パーサ (インライン専用のTokenizer/Lexer群)
インライン要素（ルビや割注）は、確定した1つのブロックテキスト（数百文字程度）を対象にするため、ここは文字単位でループを回してTokenizerとLexerの役割を細かく分離します。

class InlineParser {
    constructor(options) {
        // それぞれ単一責任を持つインライン用コンポーネント群
        this.escapeProcessor = new InlineEscapeProcessor(options);
        this.rubyProcessor = new RubyProcessor(options);
        this.emProcessor = new EmProcessor(options);
        this.warichuProcessor = new WarichuProcessor(options);
    }

    parse(text) {
        const children = [];
        let i = 0;

        while (i < text.length) {
            // 各プロセッサ（Tokenizer兼Lexerの役割）にその位置でマッチするか聞く
            
            // 1. エスケープ
            let escapeNode = this.escapeProcessor.tryProcess(text, i);
            if (escapeNode) { children.push(escapeNode.node); i = escapeNode.end; continue; }

            // 2. ルビ
            let rubyNode = this.rubyProcessor.tryProcess(text, i);
            if (rubyNode) { children.push(rubyNode.node); i = rubyNode.end; continue; }

            // 3. 強調 (em)
            let emNode = this.emProcessor.tryProcess(text, i);
            if (emNode) { children.push(emNode.node); i = emNode.end; continue; }

            // 4. 割注
            let warichuNode = this.warichuProcessor.tryProcess(text, i);
            if (warichuNode) { children.push(warichuNode.node); i = warichuNode.end; continue; }

            // 5. 通常テキスト
            let char = text[i];
            if (children.length > 0 && children[children.length - 1].type === 'Text') {
                children[children.length - 1].value += char;
            } else {
                children.push({ type: 'Text', value: char });
            }
            i++;
        }
        return children;
    }
}

class InlineEscapeProcessor {
    tryProcess(text, pos) {
        if (text[pos] === '\\' && pos + 1 < text.length) {
            return { end: pos + 2, node: { type: 'Text', value: text[pos + 1] } };
        }
        return null;
    }
}
class RubyProcessor {
    tryProcess(text, pos) {
        if (text[pos] === '｜' || text[pos] === '|') {
            const startRuby = text.indexOf('《', pos);
            const endMark = text.indexOf('》', pos);
            if (startRuby !== -1 && endMark !== -1 && startRuby < endMark) {
                return {
                    end: endMark + 1,
                    node: {
                        type: 'Ruby',
                        target: text.slice(pos + 1, startRuby),
                        ruby: text.slice(startRuby + 1, endMark)
                    }
                };
            }
        }
        return null;
    }
}
class EmProcessor {
    tryProcess(text, pos) {
        if (text[pos] === '＊' || text[pos] === '*') {
            const endEm = text.indexOf(text[pos], pos + 1);
            if (endEm !== -1) {
                return {
                    end: endEm + 1,
                    node: { type: 'Emphasis', value: text.slice(pos + 1, endEm) }

            }
        }

class WarichuProcessor {
    tryProcess(text, pos) {
        // 例: 〔上段|下段〕形式の割注判定
        if (text[pos] === '〔') {

                const lines = content.split('|'); // 縦中横や2行割注の区切り

                    end: endMark + 1,
                    node: { type: 'Warichu', top: lines[0] || '', bottom: lines[1] || '' }
                };
            }
        }
        return null;
    }
}

        super(`${message} (行: ${line}, 文字: ${char})\n対象付近: "${contextText}"`);
        this.name = 'JavelSyntaxError';
        this.line = line;
        this.char = char;
    }
}
// ASTノードのデータ構造を強制するファクトリclass JavelNode {
    static create(type, props = {}) {
        return Object.assign({ type, raw: '' }, props);
    }
}
class JavelParser {
    constructor(options = {}) {
        // デフォルト設定（Javel日本語小説仕様）
        this.options = Object.assign({
            fenceMarkers: ['＝', '`', '~', '■'],
            headingMarker: '＃',
            thematicBreakMarker: '＊＊＊',
            plugins: {
                fences: {} // 外部からフェンス処理関数をプラグイン可能
            }
        }, options);

        // インラインパーサはここで「ただ1つだけ」生成して使い回す（newしすぎ問題の解消）
        this.inlineParser = new InlineParser(this.options);
        
        this.tokenizer = new Tokenizer(this.options);
        this.lexer = new Lexer(this.options, this.inlineParser);
    }
    
    *parse(manuscript) {
        // 1. 改行コードを \n に完全統一（文字位置を狂わせない事前一括置換）
        const normalized = manuscript.replace(/\r\n/g, '\n').replace(/\r/g, '\n');

        // 2. トークナイザから上がってきたトークンを、Lexerで1つずつASTノードに変換して yield
        for (let token of this.tokenizer.tokenize(normalized)) {
            yield this.lexer.lex(token);
        }
    }
}

## 2. Tokenizers（共通化・無限ループ防止・未閉じ検知の責任）
最上位の Tokenizer はテキストのインデックス進行（カーソル）を一元管理し、改行数をベースに行番号も追跡します。

class Tokenizer {
    constructor(options) {
        this.options = options;
        this.blankTokenizer = new BlankTokenizer(options);
        this.fenceTokenizer = new FenceTokenizer(options);
        this.blockTokenizer = new BlockTokenizer(options);
    }

    // テキストから行番号を計算するヘルパー（エラー出力用）
    getLineNumber(text, pos) {
        return text.slice(0, pos).split('\n').length;
    }

    *tokenize(manuscript) {
        let pos = 0;
        const length = manuscript.length;

        while (pos < length) {
            // 1. 3連続以上の空行（Blank）の切り出し
            let blankToken = this.blankTokenizer.tryTokenize(manuscript, pos);
            if (blankToken) {
                pos = blankToken.end;
                yield blankToken;
                continue;
            }

            // 2. フェンス（通常・エスケープ）の切り出し（未閉じは内部で例外スロー）
            let fenceToken = this.fenceTokenizer.tryTokenize(manuscript, pos, this);
            if (fenceToken) {
                pos = fenceToken.end;
                yield fenceToken;
                continue;
            }

            // 3. それ以外（2連続改行、または次のフェンス/Blankの手前まで）をBlockとして切り出す
            let blockToken = this.blockTokenizer.tokenizeBlock(manuscript, pos, this);
            
            // 安全弁の修正：子Tokenizerが正しく範囲を切り出していることを前提とし、
            // 万が一カーソルが進まないバグがあれば例外を出してフリーズを防ぐ
            if (blockToken.end <= pos) {
                const line = this.getLineNumber(manuscript, pos);
                throw new JavelSyntaxError("パース中に不正な文字位置の停滞が発生しました。", line, pos, manuscript.slice(pos, pos + 20));
            }

            pos = blockToken.end;
            yield blockToken;
        }
    }
}
class BlankTokenizer {
    constructor(options) { this.options = options; }
    
    tryTokenize(manuscript, pos) {
        let current = pos;
        let newlineCount = 0;

        // スペースやタブは混ぜず、純粋に改行コードのみをカウント
        while (current < manuscript.length && manuscript[current] === '\n') {
            newlineCount++;
            current++;
        }

        // 3連続以上の改行のみを独立した「Blank」トークンとする（2連続はBlockの区切り）
        if (newlineCount >= 3) {
            return {
                type: 'Blank',
                start: pos,
                end: current,
                count: newlineCount
            };
        }
        return null;
    }
}
class FenceTokenizer {
    constructor(options) { this.options = options; }

    tryTokenize(manuscript, pos, master) {
        if (pos > 0 && manuscript[pos - 1] !== '\n') return null;

        let current = pos;
        
        // --- A. エスケープフェンスの判定 (｛｛｛ ~ ｝｝｝) ---
        if (manuscript[current] === '｛') {
            let openCount = 0;
            while (current < manuscript.length && manuscript[current] === '｛') {
                openCount++;
                current++;
            }
            // 3つ以上の「｛」かつ直後が改行、またはその行内にあるか
            if (openCount >= 3) {
                // 改行をスキップ
                if (manuscript[current] === '\n') current++;
                let contentStart = current;
                
                // 閉じ記号「｝」の連続を探す
                const closeSign = '｝'.repeat(openCount);
                let foundEnd = manuscript.indexOf(closeSign, current);
                
                if (foundEnd === -1) {
                    const line = master.getLineNumber(manuscript, pos);
                    throw new JavelSyntaxError(`エスケープフェンス「${'｛'.repeat(openCount)}」が閉じられていません。`, line, pos, manuscript.slice(pos, pos + 30));
                }
                
                let contentEnd = foundEnd;
                // 直前の改行をトリムするための調整
                if (manuscript[foundEnd - 1] === '\n') contentEnd--;

                let endPos = foundEnd + openCount;
                if (manuscript[endPos] === '\n') endPos++; // 直後の改行を消費

                return {
                    type: 'EscapeFence',
                    start: pos,
                    end: endPos,
                    raw: manuscript.slice(pos, endPos),
                    content: manuscript.slice(contentStart, contentEnd)
                };
            }
            return null;
        }

        // --- B. 通常フェンスの判定 (＝、` など) ---
        const markerChar = manuscript[current];
        if (!this.options.fenceMarkers.includes(markerChar)) return null;

        let markerCount = 0;
        while (current < manuscript.length && manuscript[current] === markerChar) {
            markerCount++;
            current++;
        }
        if (markerCount < 3) return null; // 3文字以上

        // 引数の取得（行末まで）
        let argStart = current;
        while (current < manuscript.length && manuscript[current] !== '\n') {
            current++;
        }
        const info = manuscript.slice(argStart, current).trim();
        if (manuscript[current] === '\n') current++;

        let contentStart = current;
        let openFences = 1;

        // 終了行の厳密な探索（ネスト対応）
        while (current < manuscript.length) {
            if (manuscript[current - 1] === '\n') {
                let checkPos = current;
                let closeCount = 0;
                while (checkPos < manuscript.length && manuscript[checkPos] === markerChar) {
                    closeCount++;
                    checkPos++;
                }

                if (closeCount === markerCount && (checkPos === manuscript.length || manuscript[checkPos] === '\n')) {
                    openFences--;
                    if (openFences === 0) {
                        let contentEnd = current - 1;
                        let endPos = checkPos;
                        if (manuscript[endPos] === '\n') endPos++;
                        
                        return {
                            type: 'Fence',
                            start: pos,
                            end: endPos,
                            markerChar: markerChar,
                            markerCount: markerCount,
                            info: info,
                            rawContent: manuscript.slice(contentStart, contentEnd)
                        };
                    }
                } else if (closeCount === markerCount + 1) {
                    // 記号が1つ多い場合は子フェンスとしてカウント（ネスト対応）
                    openFences++;
                }
            }
            current++;
        }

        // 閉じられずに文末に達した場合は、サイレント消滅させず構文エラーにする！
        const line = master.getLineNumber(manuscript, pos);
        throw new JavelSyntaxError(`フェンス「${markerChar.repeat(markerCount)}」が閉じられていません。`, line, pos, manuscript.slice(pos, pos + 30));
    }
}
class BlockTokenizer {
    constructor(options) { this.options = options; }

    tokenizeBlock(manuscript, pos, masterTokenizer) {
        let current = pos;
        
        while (current < manuscript.length) {
            // 行頭のタイミングで、次の要素が来ないかをチェック
            if (current === pos || manuscript[current - 1] === '\n') {
                // 3連続改行(Blank)か、フェンス(通常・エスケープ)が始まるならここで打ち切る
                if (masterTokenizer.blankTokenizer.tryTokenize(manuscript, current) ||
                    masterTokenizer.fenceTokenizer.tryTokenize(manuscript, current, masterTokenizer)) {
                    break;
                }
                
                // 2連続改行（＝ブロック要素の区切り）に達した場合も打ち切る
                if (manuscript[current] === '\n' && manuscript[current + 1] === '\n') {
                    // 次の区切り改行の手前までをこのブロックの範囲とする
                    break;
                }
            }
            current++;
        }

        return {
            type: 'Block',
            start: pos,
            end: current,
            rawText: manuscript.slice(pos, current)
        };
    }
}

## 3. Lexers（最上位ノード構築・プラグイン拡張の責任）
Lexer はトークンを受け取り、プラグイン関数を安全に呼び出す、あるいは内部パーサ（inlineParser）をコンテキスト経由でプラグインに提供する責任を持ちます。

class Lexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.inlineParser = inlineParser;
        this.fenceLexer = new FenceLexer(options, inlineParser);
        this.blankLexer = new BlankLexer(options);
        this.blockLexer = new BlockLexer(options, inlineParser);
        this.escapeFenceLexer = new EscapeFenceLexer(options);
    }

    lex(token) {
        switch (token.type) {
            case 'Blank': return this.blankLexer.lex(token);
            case 'EscapeFence': return this.escapeFenceLexer.lex(token);
            case 'Fence': return this.fenceLexer.lex(token);
            case 'Block': return this.blockLexer.lex(token);
            default: throw new Error(`未知のトークンタイプです: ${token.type}`);
        }
    }
}
class EscapeFenceLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return JavelNode.create('EscapeFence', {
            raw: token.raw,
            content: token.content
        });
    }
}
class FenceLexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.inlineParser = inlineParser;
    }
    
    lex(token) {
        // プラグインによる動的フックの呼び出しロジック
        const markerPlugins = this.options.plugins.fences[token.markerChar];
        if (markerPlugins && typeof markerPlugins[token.info] === 'function') {
            // プラグイン関数に、トークン情報と内部パーサを利用できるコンテキストを渡す
            const context = {
                createNode: JavelNode.create,
                parseInline: (text) => this.inlineParser.parse(text)
            };
            return markerPlugins[token.info](token, context);
        }

        // デフォルトのフェンスノード（原稿復元情報を完備）
        return JavelNode.create('Fence', {
            markerChar: token.markerChar,
            markerCount: token.markerCount,
            info: token.info,
            content: token.rawContent
        });
    }
}
class BlankLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return JavelNode.create('Blank', { count: token.count });
    }
}
class BlockLexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.inlineParser = inlineParser;
    }

    lex(token) {
        const text = token.rawText.trim();

        // 見出しの判定（全角 ＃ 仕様）
        if (text.startsWith(this.options.headingMarker)) {
            let level = 0;
            const marker = this.options.headingMarker;
            while (level < text.length && text[level] === marker && level < 6) {
                level++;
            }
            const innerText = text.slice(level).trim();
            return JavelNode.create('Heading', {
                level: level,
                rawMarker: marker.repeat(level),
                children: this.inlineParser.parse(innerText)
            });
        } 
        
        // 主題区切りの判定（全角 ＊＊＊ 仕様）
        if (text === this.options.thematicBreakMarker) {
            return JavelNode.create('ThematicBreak', { raw: token.rawText });
        }

        // それ以外はすべて一般の段落（インラインパースへ丸投げ）
        return JavelNode.create('Paragraph', {
            children: this.inlineParser.parse(token.rawText)
        });
    }
}

## 4. Inline（個別最適化されたインラインプロセッサとTextNode回収機構）
インライン要素は共通化せず、個別の仕様に合わせて愚直に実装しています。文字が虚空に消えるバグは完全に排除され、マッチしなかった文字は TextNode に安全に畳み込まれます。

class InlineParser {
    constructor(options) {
        this.options = options;
        // 個別に特化して実装された独立プロセッサ群
        this.processors = [
            new InlineEscapeProcessor(),
            new ParagraphBrProcessor(),
            new RubyProcessor(options),
            new EmphasisProcessor(options),
            new WarichuProcessor(options)
        ];
    }

    parse(text) {
        const children = [];
        let i = 0;

        while (i < text.length) {
            let matched = false;

            for (const processor of this.processors) {
                const result = processor.tryProcess(text, i);
                if (result) {
                    children.push(result.node);
                    i = result.end;
                    matched = true;
                    break;
                }
            }

            // 【真のインライン安全弁】どの構文にもマッチしなかった文字は1文字も消さずに TextNode で回収
            if (!matched) {
                const char = text[i];
                if (children.length > 0 && children[children.length - 1].type === 'TextNode') {
                    children[children.length - 1].raw += char;
                    children[children.length - 1].value += char;
                } else {
                    children.push(JavelNode.create('TextNode', { raw: char, value: char }));
                }
                i++;
            }
        }
        return children;
    }
}
// 1. 動的インラインエスケープ（Javel固有仕様：中身が空なら外側がテキスト化）class InlineEscapeProcessor {
    tryProcess(text, pos) {
        if (text[pos] !== '｛') return null;

        let openCount = 0;
        let current = pos;
        while (current < text.length && text[current] === '｛') {
            openCount++;
            current++;
        }

        const closeSign = '｝'.repeat(openCount);
        const endPos = text.indexOf(closeSign, current);

        // 閉じ括弧が見つからない場合は構文エラーにせず、地の文（TextNode）としてそのまま出す
        if (endPos === -1) return null;

        const content = text.slice(current, endPos);
        const totalEnd = endPos + openCount;

        // 【Javel仕様】もしエスケープ内の内部テキスト数がゼロ（空）だった場合、
        // 最も内側のエスケープメタ文字がテキストであると解釈する
        if (content.length === 0) {
            const visibleBrackets = '｛｝';
            const outerRaw = text.slice(pos, totalEnd);
            return {
                end: totalEnd,
                node: JavelNode.create('TextNode', { raw: outerRaw, value: visibleBrackets })
            };
        }

        return {
            end: totalEnd,
            node: JavelNode.create('InlineEscape', {
                raw: text.slice(pos, totalEnd),
                content: content
            })
        };
    }
}
// 2. 段落内単一改行を br 要素へclass ParagraphBrProcessor {
    tryProcess(text, pos) {
        if (text[pos] === '\n') {
            return { end: pos + 1, node: JavelNode.create('Br', { raw: '\n' }) };
        }
        return null;
    }
}
// 3. ルビプロセッサ（パイプ省略形と全角《》対応の個別実装）class RubyProcessor {
    constructor(options) { this.options = options; }
    
    tryProcess(text, pos) {
        // A. パイプあり形式 (｜漢字《かんじ》)
        if (text[pos] === '｜' || text[pos] === '|') {
            const startRuby = text.indexOf('《', pos);
            const endRuby = text.indexOf('》', startRuby);
            if (startRuby !== -1 && endRuby !== -1 && startRuby < endRuby) {
                return {
                    end: endRuby + 1,
                    node: JavelNode.create('Ruby', {
                        raw: text.slice(pos, endRuby + 1),
                        target: text.slice(pos + 1, startRuby),
                        ruby: text.slice(startRuby + 1, endRuby)
                    })
                };
            }
        }

        // B. パイプ省略形式 (漢字《かんじ》)
        if (text[pos] === '《') {
            const endRuby = text.indexOf('》', pos);
            if (endRuby === -1) return null;

            // 漢字、カタカナ、アルファベットなどのルビ対象文字を「後ろ向き」に判定して吸い出す
            let targetStart = pos - 1;
            const kanjiRegex = /[\u4e00-\u9faf\u3400-\u4dbf\u30a0-\u30ff\uff66-\uff9fA-Za-z]/;
            
            while (targetStart >= 0 && kanjiRegex.test(text[targetStart])) {
                targetStart--;
            }
            targetStart++; // マッチしなくなった境界の次

            if (targetStart < pos) { // ルビ対象文字が直前に存在した場合
                const target = text.slice(targetStart, pos);
                const ruby = text.slice(pos + 1, endRuby);
                return {
                    end: endRuby + 1,
                    node: JavelNode.create('Ruby', {
                        raw: text.slice(targetStart, endRuby + 1),
                        target: target,
                        ruby: ruby,
                        isShortForm: true // 省略形フラグ
                    })
                };
            }
        }
        return null;
    }
}
// 4. 傍点プロセッサ（；傍点；）class EmphasisProcessor {
    constructor(options) { this.options = options; }
    tryProcess(text, pos) {
        const marker = '；';
        if (!text.startsWith(marker, pos)) return null;

        const endPos = text.indexOf(marker, pos + marker.length);
        if (endPos === -1) return null;

        const content = text.slice(pos + marker.length, endPos);
        return {
            end: endPos + marker.length,
            node: JavelNode.create('Emphasis', {
                raw: text.slice(pos, endPos + marker.length),
                content: content
            })
        };
    }
}
// 5. 割注プロセッサ（〔割注〕）class WarichuProcessor {
    constructor(options) { this.options = options; }
    tryProcess(text, pos) {
        if (text[pos] !== '〔') return null;

        const endPos = text.indexOf('〕', pos);
        if (endPos === -1) return null;

        const content = text.slice(pos + 1, endPos);
        // パイプや全角｜による上下段分離をサポート
        const parts = content.split(/[｜|]/);

        return {
            end: endPos + 1,
            node: JavelNode.create('Warichu', {
                raw: text.slice(pos, endPos + 1),
                top: parts[0],
                bottom: parts[1] || null
            })
        };
    }
}

今回作成したかったJavelは、DocSynという文書構文用構文(Document Syntax)というメタ構文を具象実装することで実現する形にする。

DocSynは私が最重視した大枠の実装である。つまり一つのテキストデータを解析して構文木を返すこと。但しメモリ超過せぬようnode一つずつ返却すること。構文はフェンス、ブロック、インラインの３つに大別できること。それらはオプション設定やプラグインで可変にできること。こうした大枠だけを実装した親仕様である。

DocSynはMarkdownのような人間が目で見た時に原稿がある程度出力結果と合致しているような文書を作成する時の原稿として使う。逆にYAMLのようなデータ構造を表す構文としては不適切。人間が読んで理解しやすいテキストは文書(Document)と呼ぶ。ならば文書のための構文であればDocument Syntaxだ。それを縮めてDocSyncと仮名した。

DocSyncの構文体系はフェンス、ブロック、インラインに大別できる。フェンスはユーザが任意のパース処理を仕込める最も拡張性の高い要素である。次いで大きいのはブロック。これは2連続以上の改行でブロック間を区切るものとする。インラインはブロック内に収まるもの。つまり連続した改行が含まれることがない形で記入できるものすべてだ。

こうした仕様により、文書は大凡、見出しと段落という二つのブロックで記述することになる。これだけでもう原稿の見た目と出力結果が親しい読みやすい形となるはずだ。フェンスは拡張性、インラインは詳細表現をそれぞれ強化するものである。

パース処理を単純化するためには、パーサに入ってきたデータは既に上記を満たしている必要がある。これまでの実装では改行コードの一括修正もパース処理に含めていたが、それは責任分離ができていなかった。よって今後は文字セット、BOM、改行コードをnormalizerが担当することとする。尤も、BOMはファイルデータだけであり文字列データには存在しないので不要かもしれないが。

また、改行コードは一括置換するだけなら簡単だが、前回の実装ではそのせいでエラー時に返却する文字列の位置がズレる事態が発生していた。例えばwindowsでCR-LFな改行コードの原稿テキストをパーサに投入すると、まずnormalizerによって改行コードがLFに置換される。この時点でもうユーザの手元にある原稿とは別物になる。文字の位置も改行の数だけズレていく。よって例外発生時に文字列の位置を表示しても、ユーザの手元にあるCR-LFな改行で書かれた原稿とは別の位置を示してしまい、問題箇所の特定ができなくなってしまう。

* なぜ正規化が必要なのか？（具体例）
	* 検索・照合の精度向上：「ﾊﾟｿｺﾝ」と「パソコン」など、全角・半角の違いで検索にヒットしない、あるいはシステム上で同じデータとして扱えないトラブルを防ぎます。

1. NFC（Normalization Form C）：分解されている文字を合成する形式。MacとWindows、Webなどでのデータ交換の標準として推奨されています。
2. NFD（Normalization Form D）：文字を基底文字と結合文字に分解する形式。macOSの内部処理や特定のデータベースで利用されます。
3. NFKC（Normalization Form KC）：NFCの処理に加えて、全角・半角などの「互換文字」を標準的な文字に統合する形式。検索システムなどで最もよく使われます。
4. NFKD（Normalization Form KD）：互換文字の統合を行ったうえで、文字を分解する形式です。

// 厳格な構文エラーclass JavelSyntaxError extends Error {
    constructor(message, pos) { super(`${message} (位置: ${pos})`); this.name = 'JavelSyntaxError'; }
}
class JavelParser {
    constructor(options = {}) {
        this.options = Object.assign({
            fenceMarkers: ['＝＝＝', '```', '～～～'], // 登録するフェンス記号の塊
            headingMarker: '＃',
            thematicBreakMarker: '＊＊＊',
            plugins: { fences: {} }
        }, options);

        // インラインパースは配列で定義してDRYにループを回す
        this.inlineRules = [
            // A. インラインエスケープ： ｛ ｝ のネスト対応
            {
                tryProcess: (text, pos) => {
                    if (text[pos] !== '｛') return null;
                    let open = 0, current = pos;
                    while (current < text.length && text[current] === '｛') { open++; current++; }
                    const closeSign = '｝'.repeat(open);
                    const endPos = text.indexOf(closeSign, current);
                    if (endPos === -1) return null; // 閉じがなければ普通の文字としてスルー
                    const content = text.slice(current, endPos);
                    const totalEnd = endPos + open;
                    // 中身が空なら、最も内側の括弧ペアをテキスト化
                    return {
                        end: totalEnd,
                        node: { type: 'InlineEscape', raw: text.slice(pos, totalEnd), content: content || '｛｝' }
                    };
                }
            },
            // B. ルビ（仮実装・｜必須の形）
            {
                tryProcess: (text, pos) => {
                    if (text[pos] !== '｜' && text[pos] !== '|') return null;
                    const match = text.slice(pos).match(/^([｜|])([^《]+)《([^》]+)》/);
                    if (!match) return null;
                    return {
                        end: pos + match[0].length,
                        node: { type: 'Ruby', raw: match[0], target: match[2], ruby: match[3] }
                    };
                }
            },
            // C. 傍点・割注（仮実装・正規表現でサクッと）
            {
                tryProcess: (text, pos) => {
                    const match = text.slice(pos).match(/^；([^；]+)；/); // 傍点
                    if (!match) return null;
                    return { end: pos + match[0].length, node: { type: 'Emphasis', raw: match[0], content: match[1] } };
                }
            },
            {
                tryProcess: (text, pos) => {
                    const match = text.slice(pos).match(/^〔([^〕]+)〕/); // 割注
                    if (!match) return null;
                    return { end: pos + match[0].length, node: { type: 'Warichu', raw: match[0], content: match[1] } };
                }
            },
            // D. 改行
            {
                tryProcess: (text, pos) => {
                    if (text[pos] === '\n') return { end: pos + 1, node: { type: 'Br', raw: '\n' } };
                    return null;
                }
            }
        ];
    }

    // インライン用DRY呼び出し機構
    parseInline(text) {
        const children = [];
        let i = 0;
        while (i < text.length) {
            let matched = false;
            for (const rule of this.inlineRules) {
                const result = rule.tryProcess(text, i);
                if (result) { children.push(result.node); i = result.end; matched = true; break; }
            }
            if (!matched) {
                const char = text[i];
                if (children.length > 0 && children[children.length - 1].type === 'TextNode') {
                    children[children.length - 1].raw += char;
                    children[children.length - 1].value += char;
                } else {
                    children.push({ type: 'TextNode', raw: char, value: char });
                }
                i++;
            }
        }
        return children;
    }

    // ブロック（段落・見出し・主題区切り）パースのDRY化
    parseBlock(text) {
        const trimmed = text.trim();
        if (trimmed.startsWith(this.options.headingMarker)) {
            let level = 0;
            while (level < trimmed.length && trimmed[level] === this.options.headingMarker) level++;
            return { type: 'Heading', level, raw: text, children: this.parseInline(trimmed.slice(level).trim()) };
        }
        if (trimmed === this.options.thematicBreakMarker) {
            return { type: 'ThematicBreak', raw: text };
        }
        return { type: 'Paragraph', raw: text, children: this.parseInline(text) };
    }

    // 最上位ジェネレータ
    *parse(manuscript) {
        // 改行コードの統一
        const normalized = manuscript.replace(/\r\n/g, '\n').replace(/\r/g, '\n');
        let pos = 0;

        while (pos < normalized.length) {
            // 1. 空行（3連続以上の改行）の切り出し
            let newlineCount = 0;
            let current = pos;
            while (current < normalized.length && normalized[current] === '\n') { newlineCount++; current++; }
            if (newlineCount >= 3) {
                yield { type: 'Blank', count: newlineCount, raw: '\n'.repeat(newlineCount) };
                pos = current;
                continue;
            }

            // 2. フェンスの切り出し
            let isFence = false;
            for (const marker of this.options.fenceMarkers) {
                if ((pos === 0 || normalized[pos - 1] === '\n') && normalized.startsWith(marker, pos)) {
                    isFence = true;
                    // 引数の取得
                    let argStart = pos + marker.length;
                    let argEnd = normalized.indexOf('\n', argStart);
                    if (argEnd === -1) argEnd = normalized.length;
                    const info = normalized.slice(argStart, argEnd).trim();
                    
                    // 閉じフェンスの探索
                    const contentStart = argEnd === normalized.length ? argEnd : argEnd + 1;
                    const endFencePos = normalized.indexOf('\n' + marker, contentStart);
                    
                    if (endFencePos === -1) {
                        throw new JavelSyntaxError(`フェンス「${marker}」が閉じられていません。`, pos);
                    }

                    const content = normalized.slice(contentStart, endFencePos);
                    let totalEnd = endFencePos + 1 + marker.length;
                    if (normalized[totalEnd] === '\n') totalEnd++; // 直後の改行を消費

                    const token = { type: 'Fence', marker, info, rawContent: content };
                    pos = totalEnd;

                    // --- フェンスのLexer/プラグイン処理 ---
                    if (info === '') {
                        // 引数なし「＝＝＝」はエスケープフェンス（中身をインライン解析しない）
                        yield { type: 'EscapeFence', marker, content };
                    } else {
                        // 引数ありはプラグインをチェック
                        const plugin = this.options.plugins.fences[marker]?.[info];
                        if (typeof plugin === 'function') {
                            // プラグイン側にすべての解析能力を context として授ける
                            const context = {
                                parseInline: (t) => this.parseInline(t),
                                parseBlock: (t) => this.parseBlock(t)
                            };
                            yield plugin(token, context);
                        } else {
                            yield { type: 'Fence', marker, info, content };
                        }
                    }
                    break;
                }
            }
            if (isFence) continue;

            // 3. 通常ブロック（2連続改行、または次のフェンスの手前までをただ切り出す）
            let blockEnd = pos;
            while (blockEnd < normalized.length) {
                if (blockEnd === pos || normalized[blockEnd - 1] === '\n') {
                    // 次のフェンスのチェック
                    let hitFence = false;
                    for (const m of this.options.fenceMarkers) {
                        if (normalized.startsWith(m, blockEnd)) { hitFence = true; break; }
                    }
                    // 3連続改行（Blank）のチェック
                    if (normalized.startsWith('\n\n\n', blockEnd)) hitFence = true;
                    
                    if (hitFence) break;

                    // 2連続改行（ブロック区切り）のチェック
                    if (normalized.startsWith('\n\n', blockEnd)) break;
                }
                blockEnd++;
            }

            const rawText = normalized.slice(pos, blockEnd);
            if (rawText.length === 0) { pos++; continue; } // 無限ループ防止

            yield this.parseBlock(rawText);
            pos = blockEnd;
        }
    }
}

## Javelパーサ 厳格なるSRP完全アーキテクチャ## 1. 最上位レイヤー (JavelParser)

class JavelParser {
    constructor(options = {}) {
        this.options = Object.assign({
            fenceMarkers: ['＝＝＝', '```', '～～～'],
            headingMarker: '＃',
            thematicBreakMarker: '＊＊＊',
            plugins: { fences: {} }
        }, options);

        // 各階層のインフラを1つだけ生成して使い回す（newしすぎ問題の解消）
        this.inlineParser = new InlineParser(this.options);
        this.blockLexer = new BlockLexer(this.options, this.inlineParser);
        this.fenceLexer = new FenceLexer(this.options, this.inlineParser);
        this.blankLexer = new BlankLexer(this.options);

        // トップレベルのTokenizer群
        this.tokenizers = [
            new BlankTokenizer(this.options),
            new FenceTokenizer(this.options),
            new BlockTokenizer(this.options)
        ];
    }

    *parse(manuscript) {
        // OSごとの改行コードを \n に完全統一（文字位置を狂わせない事前一括置換）
        const normalized = manuscript.replace(/\r\n/g, '\n').replace(/\r/g, '\n');
        let pos = 0;
        const length = normalized.length;

        while (pos < length) {
            let matched = false;

            for (const tokenizer of this.tokenizers) {
                const token = tokenizer.tryTokenize(normalized, pos);
                if (token) {
                    pos = token.end;
                    // Lexerへ委譲してASTノードを生成し、1つずつ即座に返却
                    const node = this.lexTopLevel(token);
                    if (node) yield node;
                    matched = true;
                    break;
                }
            }

            // 厳格な安全弁：Tokenizerのバグでカーソルが止まった場合は即座に例外を投げる
            if (!matched) {
                throw new Error(`[JavelParser] パース不能な文字による位置の停滞が発生しました(pos: ${pos})`);
            }
        }
    }

    lexTopLevel(token) {
        switch (token.type) {
            case 'Blank': return this.blankLexer.lex(token);
            case 'Fence': return this.fenceLexer.lex(token);
            case 'Block': return this.blockLexer.lex(token);
            default: return null;
        }
    }
}

## 2. ブロック/フェンス レイヤー（Tokenizer & Lexerの分離）
各クラスは「範囲の切り出し」か「意味の解釈」のどちらか片方しか行いません。

// --- 空行 (Blank) 系統 ---class BlankTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        let current = pos;
        while (current < text.length && text[current] === '\n') { current++; }
        const count = current - pos;
        // 3連続以上の改行のみを独立したBlankブロックとして認識（2連続はブロック区切り）
        if (count >= 3) {
            return { type: 'Blank', start: pos, end: current, count };
        }
        return null;
    }
}class BlankLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return { type: 'Blank', count: token.count, raw: '\n'.repeat(token.count) };
    }
}
// --- フェンス (Fence) 系統 ---class FenceTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (pos > 0 && text[pos - 1] !== '\n') return null;

        for (const marker of this.options.fenceMarkers) {
            if (!text.startsWith(marker, pos)) continue;

            const argStart = pos + marker.length;
            let argEnd = text.indexOf('\n', argStart);
            if (argEnd === -1) argEnd = text.length;
            const info = text.slice(argStart, argEnd).trim();

            const contentStart = argEnd === text.length ? argEnd : argEnd + 1;
            const endFencePos = text.indexOf('\n' + marker, contentStart);

            // 【厳格な安全弁】閉じられていないフェンスは即座に例外
            if (endFencePos === -1) {
                throw new Error(`[JavelSyntaxError] フェンス「${marker}」が閉じられていません。(開始位置: ${pos})`);
            }

            let totalEnd = endFencePos + 1 + marker.length;
            if (text[totalEnd] === '\n') totalEnd++; // 直後の改行を消費

            return {
                type: 'Fence',
                start: pos,
                end: totalEnd,
                marker,
                info,
                rawContent: text.slice(contentStart, endFencePos)
            };
        }
        return null;
    }
}class FenceLexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.inlineParser = inlineParser;
        this.escapeLexer = new FenceEscapeLexer(options);
    }
    lex(token) {
        // 引数なし「＝＝＝」はエスケープフェンス
        if (token.info === '') {
            return { type: 'EscapeFence', marker: token.marker, content: token.rawContent };
        }

        // 引数ありの場合はプラグインの登録を確認
        const plugin = this.options.plugins.fences[token.marker]?.[token.info];
        if (typeof plugin === 'function') {
            // 将来プラグイン側で内部パース能力を再利用できるよう、コンテキストとして環境を授ける
            const context = {
                parseInline: (t) => this.inlineParser.parse(t),
                // 必要に応じて他のLexerやパーサの関数もここに安全にバインドして渡せる拡張性
                parseBlockText: (t) => {
                    const blockLexer = new BlockLexer(this.options, this.inlineParser);
                    return blockLexer.lex({ type: 'Block', rawText: t });
                }
            };
            return plugin(token, context);
        }

        // プラグイン未登録時のデフォルト挙動
        return {
            type: 'Fence',
            marker: token.marker,
            info: token.info,
            content: this.escapeLexer.lex(token.rawContent, token.marker)
        };
    }
}class FenceEscapeLexer {
    constructor(options) { this.options = options; }
    lex(text, marker) { return text; } // 将来、フェンス記号自体のエスケープ解除処理をここに実装
}
// --- 通常ブロック (Block) 系統 ---class BlockTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        let current = pos;
        while (current < text.length) {
            if (current === pos || text[current - 1] === '\n') {
                // 3連続改行（Blank）の手前なら打ち切る
                if (text.startsWith('\n\n\n', current)) break;
                // 他のフェンスの開始が来たら打ち切る
                let hitFence = false;
                for (const m of this.options.fenceMarkers) {
                    if (text.startsWith(m, current)) { hitFence = true; break; }
                }
                if (hitFence) break;
                // 2連続改行（ブロック要素区切り）に達したら打ち切る
                if (text.startsWith('\n\n', current)) break;
            }
            current++;
        }
        if (current === pos) return null;
        return { type: 'Block', start: pos, end: current, rawText: text.slice(pos, current) };
    }
}class BlockLexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.headingLexer = new HeadingLexer(options, inlineParser);
        this.thematicBreakLexer = new ThematicBreakLexer(options);
        this.paragraphLexer = new ParagraphLexer(options, inlineParser);
    }
    lex(token) {
        const trimmed = token.rawText.trim();
        if (trimmed.startsWith(this.options.headingMarker)) {
            return this.headingLexer.lex(token.rawText);
        }
        if (trimmed === this.options.thematicBreakMarker) {
            return this.thematicBreakLexer.lex(token.rawText);
        }
        return this.paragraphLexer.lex(token.rawText);
    }
}class HeadingLexer {
    constructor(options, inlineParser) { this.options = options; this.inlineParser = inlineParser; }
    lex(rawText) {
        const trimmed = rawText.trim();
        let level = 0;
        while (level < trimmed.length && trimmed[level] === this.options.headingMarker) level++;
        return {
            type: 'Heading',
            level,
            raw: rawText,
            children: this.inlineParser.parse(trimmed.slice(level).trim())
        };
    }
}class ThematicBreakLexer {
    constructor(options) { this.options = options; }
    lex(rawText) { return { type: 'ThematicBreak', raw: rawText }; }
}class ParagraphLexer {
    constructor(options, inlineParser) { this.options = options; this.inlineParser = inlineParser; }
    lex(rawText) {
        return { type: 'Paragraph', raw: rawText, children: this.inlineParser.parse(rawText) };
    }
}

## 3. インライン レイヤー（完全なる独立コンポーネント化）
インライン要素は共通の親を持たせず、それぞれが Tokenizer と Lexer の責任を持った完全な独立コンポーネントとして設計します。InlineParser はこれらを単に順番に呼び出す「ディスパッチャ（配行器）」の役割のみを担います。

class InlineParser {
    constructor(options) {
        this.options = options;
        // 将来詳細なロジックを流し込めるよう、完全に独立したコンポーネントとして配列化（DRY呼び出し）
        this.components = [
            new InlineEscapeComponent(options),
            new ParagraphBrComponent(options),
            new RubyComponent(options),
            new EmphasisComponent(options),
            new WarichuComponent(options)
        ];
        this.textNodeComponent = new TextNodeComponent(options);
    }

    parse(text) {
        const children = [];
        let i = 0;

        while (i < text.length) {
            let matched = false;

            for (const component of this.components) {
                // Tokenizerとしての役割：範囲の切り出し
                const token = component.tryTokenize(text, i);
                if (token) {
                    // Lexerとしての役割：Nodeへの変換（同クラス内で完結、または内部委譲）
                    const node = component.lex(token);
                    children.push(node);
                    i = token.end;
                    matched = true;
                    break;
                }
            }

            // 【インラインの真の安全弁】マッチしなかった残余テキストは、TextNodeComponentが一文字も漏らさず回収
            if (!matched) {
                const char = text[i];
                const token = this.textNodeComponent.tryTokenize(char, 0);
                const node = this.textNodeComponent.lex(token);

                // 隣り合うTextNode同士を綺麗に結合してASTをスッキリさせる
                if (children.length > 0 && children[children.length - 1].type === 'TextNode') {
                    children[children.length - 1].raw += node.raw;
                    children[children.length - 1].value += node.value;
                } else {
                    children.push(node);
                }
                i++;
            }
        }
        return children;
    }
}

class InlineEscapeComponent {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] !== '｛') return null;
        let open = 0, current = pos;
        while (current < text.length && text[current] === '｛') { open++; current++; }
        const closeSign = '｝'.repeat(open);
        const endPos = text.indexOf(closeSign, current);
        if (endPos === -1) return null; // 閉じがなければ構文エラーにせず、地の文としてスルー
        
        return {
            type: 'InlineEscape',
            start: pos,
            end: endPos + open,
            content: text.slice(current, endPos),
            openCount: open
        };
    }
    lex(token) {
        // 中身の文字数がゼロなら、最も内側のエスケープ文字がテキスト化される仕様の表現
        const value = token.content.length === 0 ? '｛｝' : token.content;
        return { type: 'InlineEscape', raw: '｛'.repeat(token.openCount) + token.content + '｝'.repeat(token.openCount), content: value };
    }
}
class ParagraphBrComponent {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] === '\n') return { type: 'Br', start: pos, end: pos + 1 };
        return null;
    }
    lex(token) { return { type: 'Br', raw: '\n' }; }
}
class RubyComponent {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        // 将来、ここに「｜省略形（青空文庫形式）」などの複雑な後ろ向きスキャンを実装する。
        // 現在は大枠をがっちり固定するための仮実装の定義。

        const remain = text.slice(pos);
        const match = remain.match(/^([｜|])([^《]+)《([^》]+)》/);
        if (!match) return null;
        return { type: 'Ruby', start: pos, end: pos + match[0].length, match };
    }
    lex(token) {
        return { type: 'Ruby', raw: token.match[0], target: token.match[2], ruby: token.match[3] };
    }
}
class EmphasisComponent {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (!text.startsWith('；', pos)) return null;
        const endPos = text.indexOf('；', pos + 1);
        if (endPos === -1) return null;
        return { type: 'Emphasis', start: pos, end: endPos + 1, content: text.slice(pos + 1, endPos) };
    }
    lex(token) {
        return { type: 'Emphasis', raw: `；${token.content}；`, content: token.content };
    }
}
class WarichuComponent {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] !== '〔') return null;
        const endPos = text.indexOf('〕', pos);
        if (endPos === -1) return null;
        return { type: 'Warichu', start: pos, end: endPos + 1, content: text.slice(pos + 1, endPos) };
    }
    lex(token) {
        return { type: 'Warichu', raw: `〔${token.content}〕`, content: token.content };
    }
}
class TextNodeComponent {
    constructor(options) { this.options = options; }
    tryTokenize(char, pos) {
        return { type: 'TextNode', start: pos, end: pos + 1, char };
    }
    lex(token) {
        return { type: 'TextNode', raw: token.char, value: token.char };
    }
}

   1. Tokenizer（Blank / Fence / Block）:
   テキストの先頭（pos）を受け取り、自分の担当する構文の「開始から終了までの範囲（start, end）」を特定してトークンオブジェクトを返すことだけに責任を持ちます。中身の解釈やASTノードの組み立ては一切行いません。
   2. Lexer（Blank / Fence / Block）:

   3. InlineParser:
