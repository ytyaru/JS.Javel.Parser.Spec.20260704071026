## Javelパーサ SRP完全厳守アーキテクチャ## 1. トップレベルパーサ（JavelParser）
全体の統括、改行コードの統一、そしてストリーミング（yield）の制御のみを行います。

class JavelParser {
    constructor(options = {}) {
        this.options = Object.assign({
            fenceMarkers: ['＝＝＝', '```', '～～～'],
            headingMarker: '＃',
            thematicBreakMarker: '＊＊＊',
            plugins: { fences: {} }
        }, options);

        // 各階層のインフラを独立して生成
        this.inlineParser = new InlineParser(this.options);
        
        // トップレベルのTokenizer群
        this.tokenizers = [
            new BlankTokenizer(this.options),
            new FenceTokenizer(this.options),
            new BlockTokenizer(this.options)
        ];

        // トップレベルのLexer群
        this.blankLexer = new BlankLexer(this.options);
        this.fenceLexer = new FenceLexer(this.options, this.inlineParser);
        this.blockLexer = new BlockLexer(this.options, this.inlineParser);
    }

    *parse(manuscript) {
        // 改行コードを一括統一（位置がズレないため、かつフェンス内も含めて統一してOKな処理）
        const normalized = manuscript.replace(/\r\n/g, '\n').replace(/\r/g, '\n');
        let pos = 0;
        const length = normalized.length;

        while (pos < length) {
            let matched = false;

            for (const tokenizer of this.tokenizers) {
                const token = tokenizer.tryTokenize(normalized, pos);
                if (token) {
                    pos = token.end;
                    const node = this.lexTopLevel(token);
                    if (node) yield node;
                    matched = true;
                    break;
                }
            }

            // 厳格な安全弁
            if (!matched) {
                throw new Error(`[JavelParser] パース不能な文字による位置の停滞が発生しました(pos: ${pos})`);
            }
        }
    }

    lexTopLevel(token) {
        switch (token.type) {
            case 'Blank': return this.blankLexer.lex(token);
            case 'Fence': return this.fenceLexer.lex(token);
            case 'Block': return this.blockLexer.lex(token);
            default: return null;
        }
    }
}

// ==========================================// Blank (空行) 系統// ==========================================class BlankTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        let current = pos;
        while (current < text.length && text[current] === '\n') { current++; }
        const count = current - pos;
        if (count >= 3) {
            return { type: 'Blank', start: pos, end: current, count };
        }
        return null;
    }
}class BlankLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return { type: 'Blank', count: token.count, raw: '\n'.repeat(token.count) };
    }
}
// ==========================================// Fence 系統// ==========================================class FenceTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (pos > 0 && text[pos - 1] !== '\n') return null;

        for (const marker of this.options.fenceMarkers) {
            if (!text.startsWith(marker, pos)) continue;

            const argStart = pos + marker.length;
            let argEnd = text.indexOf('\n', argStart);
            if (argEnd === -1) argEnd = text.length;
            const info = text.slice(argStart, argEnd).trim();

            const contentStart = argEnd === text.length ? argEnd : argEnd + 1;
            const endFencePos = text.indexOf('\n' + marker, contentStart);

            // 【厳格な安全弁】閉じられていないフェンスは即座に例外
            if (endFencePos === -1) {
                throw new Error(`[JavelSyntaxError] フェンス「${marker}」が閉じられていません。(開始位置: ${pos})`);
            }

            let totalEnd = endFencePos + 1 + marker.length;
            if (text[totalEnd] === '\n') totalEnd++;

            return {
                type: 'Fence',
                start: pos,
                end: totalEnd,
                marker,
                info,
                rawContent: text.slice(contentStart, endFencePos)
            };
        }
        return null;
    }
}class FenceLexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.inlineParser = inlineParser;
    }
    lex(token) {
        // 引数なしは「地の文エスケープフェンス」
        if (token.info === '') {
            return { type: 'EscapeFence', marker: token.marker, content: token.rawContent };
        }

        // プラグイン呼び出し
        const plugin = this.options.plugins.fences[token.marker]?.[token.info];
        if (typeof plugin === 'function') {
            const context = {
                parseInline: (t) => this.inlineParser.parse(t),
                parseBlockText: (t) => {
                    const blockLexer = new BlockLexer(this.options, this.inlineParser);
                    return blockLexer.lex({ type: 'Block', rawText: t });
                }
            };
            return plugin(token, context);
        }

        return { type: 'Fence', marker: token.marker, info: token.info, content: token.rawContent };
    }
}
// ==========================================// Block (段落・見出し・主題区切り) 系統// ==========================================class BlockTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        let current = pos;
        while (current < text.length) {
            if (current === pos || text[current - 1] === '\n') {
                if (text.startsWith('\n\n\n', current)) break;
                let hitFence = false;
                for (const m of this.options.fenceMarkers) {
                    if (text.startsWith(m, current)) { hitFence = true; break; }
                }
                if (hitFence) break;
                if (text.startsWith('\n\n', current)) break;
            }
            current++;
        }
        if (current === pos) return null;
        return { type: 'Block', start: pos, end: current, rawText: text.slice(pos, current) };
    }
}class BlockLexer {
    constructor(options, inlineParser) {
        this.options = options;
        this.headingLexer = new HeadingLexer(options, inlineParser);
        this.thematicBreakLexer = new ThematicBreakLexer(options);
        this.paragraphLexer = new ParagraphLexer(options, inlineParser);
    }
    lex(token) {
        const trimmed = token.rawText.trim();
        if (trimmed.startsWith(this.options.headingMarker)) {
            return this.headingLexer.lex(token.rawText);
        }
        if (trimmed === this.options.thematicBreakMarker) {
            return this.thematicBreakLexer.lex(token.rawText);
        }
        return this.paragraphLexer.lex(token.rawText);
    }
}class HeadingLexer {
    constructor(options, inlineParser) { this.options = options; this.inlineParser = inlineParser; }
    lex(rawText) {
        const trimmed = rawText.trim();
        let level = 0;
        while (level < trimmed.length && trimmed[level] === this.options.headingMarker) level++;
        return {
            type: 'Heading',
            level,
            raw: rawText,
            children: this.inlineParser.parse(trimmed.slice(level).trim())
        };
    }
}class ThematicBreakLexer {
    constructor(options) { this.options = options; }
    lex(rawText) { return { type: 'ThematicBreak', raw: rawText }; }
}class ParagraphLexer {
    constructor(options, inlineParser) { this.options = options; this.inlineParser = inlineParser; }
    lex(rawText) {
        return { type: 'Paragraph', raw: rawText, children: this.inlineParser.parse(rawText) };
    }
}

## 3. インライン レイヤー（すべての要素でTokenizer/Lexerを徹底分離）
InlineParser はただのディスパッチャであり、中身の切り出しと翻訳はそれぞれ対になる Tokenizer と Lexer に完全に分離されています。

class InlineParser {
    constructor(options) {
        this.options = options;

        // 全インライン要素の Tokenizer と Lexer を完全に分離して対で管理する（DRY機構）
        this.rules = [
            { tokenizer: new InlineEscapeTokenizer(options), lexer: new InlineEscapeLexer(options) },
            { tokenizer: new ParagraphBrTokenizer(options),   lexer: new ParagraphBrLexer(options) },
            { tokenizer: new RubyTokenizer(options),          lexer: new RubyLexer(options) },
            { tokenizer: new EmphasisTokenizer(options),      lexer: new EmphasisLexer(options) },
            { tokenizer: new WarichuTokenizer(options),       lexer: new WarichuLexer(options) }
        ];

        this.textNodeTokenizer = new TextNodeTokenizer(options);
        this.textNodeLexer = new TextNodeLexer(options);
    }

    parse(text) {
        const children = [];
        let i = 0;

        while (i < text.length) {
            let matched = false;

            for (const rule of this.rules) {
                // 1. Tokenizerによる「範囲の切り出し」のみ
                const token = rule.tokenizer.tryTokenize(text, i);
                if (token) {
                    // 2. Lexerによる「ノードへの翻訳」のみ
                    const node = rule.lexer.lex(token);
                    children.push(node);
                    i = token.end;
                    matched = true;
                    break;
                }
            }

            // 【インライン安全弁】溢れたテキストの1文字ずつの回収も、TokenizerとLexerを厳密に分離
            if (!matched) {
                const char = text[i];
                const token = this.textNodeTokenizer.tryTokenize(char, 0);
                const node = this.textNodeLexer.lex(token);

                if (children.length > 0 && children[children.length - 1].type === 'TextNode') {
                    children[children.length - 1].raw += node.raw;
                    children[children.length - 1].value += node.value;
                } else {
                    children.push(node);
                }
                i++;
            }
        }
        return children;
    }
}
// ==========================================// InlineEscape (インラインエスケープ ｛ ｝) 系統// ==========================================class InlineEscapeTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] !== '｛') return null;
        let open = 0, current = pos;
        while (current < text.length && text[current] === '｛') { open++; current++; }
        const closeSign = '｝'.repeat(open);
        const endPos = text.indexOf(closeSign, current);
        if (endPos === -1) return null; // 未閉じは地の文にする
        
        return {
            type: 'InlineEscape',
            start: pos,
            end: endPos + open,
            content: text.slice(current, endPos),
            openCount: open
        };
    }
}class InlineEscapeLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        const value = token.content.length === 0 ? '｛｝' : token.content;
        return {
            type: 'InlineEscape',
            raw: '｛'.repeat(token.openCount) + token.content + '｝'.repeat(token.openCount),
            content: value
        };
    }
}
// ==========================================// ParagraphBr (段落内改行) 系統// ==========================================class ParagraphBrTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] === '\n') return { type: 'Br', start: pos, end: pos + 1 };
        return null;
    }
}class ParagraphBrLexer {
    constructor(options) { this.options = options; }
    lex(token) { return { type: 'Br', raw: '\n' }; }
}
// ==========================================// Ruby (ルビ) 系統// ==========================================class RubyTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] !== '｜' && text[pos] !== '|') return null;
        const remain = text.slice(pos);
        const match = remain.match(/^([｜|])([^《]+)《([^》]+)》/);
        if (!match) return null;
        return { type: 'Ruby', start: pos, end: pos + match.length, match };
    }
}class RubyLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return { type: 'Ruby', raw: token.match, target: token.match, ruby: token.match };
    }
}
// ==========================================// Emphasis (傍点) 系統// ==========================================class EmphasisTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (!text.startsWith('；', pos)) return null;
        const endPos = text.indexOf('；', pos + 1);
        if (endPos === -1) return null;
        return { type: 'Emphasis', start: pos, end: endPos + 1, content: text.slice(pos + 1, endPos) };
    }
}class EmphasisLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return { type: 'Emphasis', raw: `；${token.content}；`, content: token.content };
    }
}
// ==========================================// Warichu (割注) 系統// ==========================================class WarichuTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(text, pos) {
        if (text[pos] !== '〔') return null;
        const endPos = text.indexOf('〕', pos);
        if (endPos === -1) return null;
        return { type: 'Warichu', start: pos, end: endPos + 1, content: text.slice(pos + 1, endPos) };
    }
}class WarichuLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return { type: 'Warichu', raw: `〔${token.content}〕`, content: token.content };
    }
}
// ==========================================// TextNode (プレーンテキスト回収) 系統// ==========================================class TextNodeTokenizer {
    constructor(options) { this.options = options; }
    tryTokenize(char, pos) {
        return { type: 'TextNode', start: pos, end: pos + 1, char };
    }
}class TextNodeLexer {
    constructor(options) { this.options = options; }
    lex(token) {
        return { type: 'TextNode', raw: token.char, value: token.char };
    }
}

## 原則の厳守証明

   1. Tokenizer クラス（全10クラス）
   役割は、文字列内の特定の位置から「自分の形にマッチする範囲を切り出し、境界点（start, end）と最小限の生メタデータを含むトークンオブジェクトを生成して返す」ことだけです。AST構造の決定やパースデータの決定（意味の抽出）には1ミリも関与しません。
   2. Lexer クラス（全10クラス）
   役割は、完成したトークンオブジェクトを受け取り、「対応するASTノード（JFS / JSON構造オブジェクト）に翻訳して出力する」ことだけです。文字列の探索やスキャン、スライス範囲の計算は一切行いません。
